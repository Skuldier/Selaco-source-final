CCMD(archipelago_connect)
{
    if (argv.argc() < 4) {
        Printf("Usage: archipelago_connect <host> <port> <slot_name> [password]\n");
        Printf("Example: archipelago_connect archipelago.gg 38281 MySlot\n");
        return;
    }
    
    const char* host = argv[1];
    int port = atoi(argv[2]);
    const char* slot_name = argv[3];
    const char* password = argv.argc() > 4 ? argv[4] : "";
    
    CMD_ArchipelagoConnect(host, port, slot_name, password);
}

CCMD(archipelago_disconnect)
{
    CMD_ArchipelagoDisconnect();
}

CCMD(archipelago_status) 
{
    CMD_ArchipelagoStatus();
}

{
    CMD_ArchipelagoTest();
}
